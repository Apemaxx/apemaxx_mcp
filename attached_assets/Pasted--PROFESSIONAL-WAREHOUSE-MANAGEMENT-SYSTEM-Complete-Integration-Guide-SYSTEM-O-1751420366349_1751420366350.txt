  switch (action) {
    case 'create':
      setLocation('/warehouse?mode=create');
      break;
    case 'view':
      if (receiptId) {
        setLocation(`/warehouse?receipt=${receiptId}`);
      } else {
        setLocation('/warehouse');
      }
      break;
    case 'manage':
      setLocation('/warehouse');
      break;
    default:
      setLocation('/warehouse');
  }
};
